```c++
class Solution {
public:
    string reverseVowels(string s) {
        unordered_set<char> vowels = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
        int left = 0;
        int right = s.size()-1;
        
        while (left < right) {
            while (left < s.size() && vowels.find(s[left]) == vowels.end()) {
                left++;
            }
           
            if (left == s.size()) {
                break;
            }
            while (right >= 0 && vowels.find(s[right]) == vowels.end()) {
                right--;
            }
            if (left < right) { 
                swap(s[left], s[right]);
                left++;
                right--;
            }
        }
        return s;
    }
};
```
